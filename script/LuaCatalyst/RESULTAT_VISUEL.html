📖 Comment utiliser dans Roblox
-- Charger la librairie local ScreenBuilder = loadstring(readfile("script/LuaCatalyst/libs/ScreenBuilder.lua"))() -- Créer l'éditeur avec coloration syntaxique local editorFrame, textBox = ScreenBuilder.CodeEditor.Create({ Parent = votreGUI, Size = UDim2.new(0, 700, 0, 500), BackgroundColor3 = Color3.fromRGB(30, 30, 30), Text = "local x = 42" }) -- La coloration s'applique automatiquement ! -- Vous pouvez aussi forcer manuellement : textBox:Highlight("lua") -- Pour Lua textBox:Highlight("json") -- Pour JSON
